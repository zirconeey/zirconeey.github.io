\chapter{四大满贯赛事}

\textbf{网球大满贯}（Grand Slam）是指一位网球选手能在以下 4 项公开赛中\textbf{全部夺冠}：

\section{四大满贯一览}

\begin{center}
\begin{tabular}{@{} >{\bfseries}l l l l @{}}
    \toprule
    赛事 & 场地 & 举办时间 & 举办地点 \\
    \midrule
    澳大利亚网球公开赛 & 硬地（慢速） & 每年 1 月 & 墨尔本（澳大利亚） \\
    法国网球公开赛 & 红土 & 每年 5--6 月 & 巴黎罗兰·加洛斯 \\
    温布尔登锦标赛 & 草地 & 每年 6--7 月 & 伦敦温布尔登 \\
    美国网球公开赛 & 硬地（快速） & 每年 8--9 月 & 纽约 \\
    \bottomrule
\end{tabular}
\end{center}

\state{各赛事地位}{
    \begin{itemize}
        \item \textbf{澳网}：网球\textbf{硬地赛事}中声望最高的 2 大赛事之一（另一个是美网）；\textbf{创办最晚}（1905）。
        \item \textbf{法网}：网球\textbf{红土赛事}中的\textbf{最高荣誉}。
        \item \textbf{温网}：网球\textbf{草地赛事}中的最高荣誉，也是\textbf{全世界声望最高、历史最久}的网球赛事。
        \item \textbf{美网}：硬地赛事中声望最高的 2 大赛事之一。
    \end{itemize}
}

\rmkb{
    \textbf{重点}：2021 年\textbf{德约科维奇}包揽前三满贯男单冠军，美网男单\textbf{梅德韦杰夫}（俄罗斯）终结其大满贯年。女单四位冠军\textbf{全部来自不同国家}——大坂直美（日本）、克雷吉茨科娃（捷克）、巴蒂（澳大利亚）、拉杜卡努（英国）。
