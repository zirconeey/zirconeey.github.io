\chapter{网球的起源与历史}

\section{起源}

\state{网球的诞生}{
    \begin{itemize}
        \item \textbf{12--13 世纪}，网球最早起源于\textbf{法国}，当时是一种"\textbf{掌中游戏}"（用手掌击球）。
        \item \textbf{1873 年}，英国\textbf{温菲尔德少校}（Major Wingfield）在\textbf{羽毛球运动}的启示下，进一步改进了早期的网球打法，把一般在土地上的活动移到了\textbf{草地}，提出了一套较为接近现代网球的打法。
    \end{itemize}
}

\section{现代网球的形成}

\state{里程碑年表}{
    \begin{itemize}
        \item \textbf{1877 年}：全英板球俱乐部在\textbf{温布尔登}举行了第一届草地网球锦标赛。以\textbf{亨利·琼斯}为首的裁判委员会草拟的比赛规则是现代网球比赛规则的基础，其中的\textbf{盘制、局制、换位法}等规则沿用至今。同年，首届\textbf{温布尔登网球公开赛}在伦敦西郊温布尔登总部举办——\textbf{是现代网球史上最早的网球比赛}。
        \item \textbf{1881 年}：美国草地网球协会成立；同年 8 月 31 日至 9 月 3 日，成功举办第一届\textbf{美国草地网球锦标赛}。首届\textbf{美国网球公开赛}在美国罗德岛新港举行。
        \item \textbf{1891 年}：第一届\textbf{法国网球公开赛}于巴黎西部罗兰·加洛斯的体育场内举行。
        \item \textbf{1905 年}：\textbf{澳大利亚网球公开赛}在墨尔本开赛——比温布尔登网球公开赛晚 \textbf{28 年}，是四大网球公开赛中\textbf{创办最晚}的赛事。
    \end{itemize}
}

\rmkb{
    \textbf{易考要点}：
    \begin{itemize}
        \item 起源国：\textbf{法国}；现代网球起源国：\textbf{英国}；
        \item 温布尔登 1877 年（最早）、美网 1881 年、法网 1891 年、澳网 1905 年（最晚）；
        \item "掌中游戏"是早期形式；温菲尔德少校受\textbf{羽毛球}启示改良。
    \end{itemize}
}
