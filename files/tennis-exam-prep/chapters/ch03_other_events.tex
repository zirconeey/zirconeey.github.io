\chapter{其他重要赛事}

\section{中国网球公开赛（中网）}

\state{中网的地位与参数}{
    \begin{itemize}
        \item \textbf{地位}：整体级别上仅次于四大满贯，与\textbf{印第安维尔斯大师赛、迈阿密大师赛、马德里大师赛}（红土）并称"\textbf{四大超级赛事}"。这四项中 \textbf{3 项是硬地}，\textbf{马德里是红土}。
        \item \textbf{举办时间}：每年 \textbf{9 月到 10 月}；
        \item \textbf{举办地点}：\textbf{北京}；
        \item \textbf{场地类型}：\textbf{硬地}。
    \end{itemize}
}

\state{2019 年中网冠军}{
    \begin{itemize}
        \item 男单：\textbf{蒂姆}（奥地利）
        \item 女单：\textbf{大坂直美}（日本）
        \item 男双：多迪格 / 波拉塞克（克罗地亚、斯洛伐克）
        \item 女双：凯宁 / 马泰克（美国）
    \end{itemize}
}
