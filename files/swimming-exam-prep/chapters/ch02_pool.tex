\chapter{游泳池与器材规格}

\section{国际标准游泳池}

\state{泳池主体尺寸}{
    \begin{itemize}
        \item \textbf{长度}：\textbf{50 米}；
        \item \textbf{宽度}：\textbf{至少 21 米}；
        \item \textbf{深度}：\textbf{1.80 米以上}；
        \item \textbf{泳道}：设\textbf{ 8 条}泳道，每条泳道宽 \textbf{2.50 米}；
        \item \textbf{分道线}：由\textbf{直径 5--10 厘米}的\textbf{单个浮标}连接而成。
    \end{itemize}
}

\state{出发台}{
    \begin{itemize}
        \item 除\textbf{仰泳}外，运动员比赛时必须站在\textbf{出发台}上出发；
        \item 出发台\textbf{高出水面 50--75 厘米}；
        \item 台面面积为 \textbf{50 $\times$ 50 厘米}。
    \end{itemize}
}

\section{两条重要的横贯旗绳}

\state{仰泳转身标志旗（旗绳）}{
    \begin{itemize}
        \item 横穿泳池的\textbf{仰泳转身标志旗}，应悬挂在\textbf{水面以上不低于 1.8 米、不高于 2.5 米}的位置；
        \item \textbf{距离每端池壁 5.0 米}。
    \end{itemize}
}

\state{出发犯规召回线}{
    \begin{itemize}
        \item 出发犯规召回线应悬挂在\textbf{水面以上不低于 1.2 米}的位置；
        \item \textbf{距离每端池壁 15.0 米}。
    \end{itemize}
}

    \textbf{易考点}：
    \begin{itemize}
        \item 泳池长 \textbf{50 m}、宽 \textbf{$\ge$ 21 m}、深 \textbf{$\ge$ 1.80 m}；
        \item \textbf{8 条}泳道，每道宽 \textbf{2.50 m}；分道线浮标直径 \textbf{5--10 cm}；
        \item 出发台高出水面 \textbf{50--75 cm}，台面 \textbf{50 $\times$ 50 cm}（仰泳\textbf{不用}出发台）；
        \item \textbf{仰泳转身旗绳}：水面以上 \textbf{1.8--2.5 m}，\textbf{距池壁 5 m}；
        \item \textbf{出发犯规召回线}：水面以上 \textbf{$\ge$ 1.2 m}，\textbf{距池壁 15 m}。
    \end{itemize}
