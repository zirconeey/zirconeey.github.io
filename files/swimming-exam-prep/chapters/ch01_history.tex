\chapter{游泳的奥运会历史与组织}

\section{奥运会游泳项目的演变}

\state{历届奥运会中游泳项目的变化}{
    \begin{itemize}
        \item \textbf{1896 年}——第 \textbf{1} 届奥运会（雅典）：游泳被列为奥运会竞赛项目。当时\textbf{不分泳姿}，是真正的"自由式"，只有 \textbf{100 米、500 米、1200 米} 3 个项目。
        \item \textbf{1900 年}——第 \textbf{2} 届奥运会：将\textbf{仰泳}分出。
        \item \textbf{1904 年}——第 \textbf{3} 届奥运会：又分出\textbf{蛙泳}。
        \item \textbf{1956 年}——第 \textbf{16} 届奥运会：增加了\textbf{蝶泳}。至此\textbf{定型为 4 种泳姿}（自由泳、仰泳、蛙泳、蝶泳）。
    \end{itemize}
}

\state{现代奥运会游泳项目结构}{
    此后的奥运会游泳比赛发展到共有：
    \begin{itemize}
        \item \textbf{6 大项}：自由泳、蛙泳、蝶泳、仰泳、\textbf{混合泳}、\textbf{接力}（自由泳与混合泳）；
        \item 共计\textbf{ 32 个小项}；
        \item 是奥运会中\textbf{仅次于田径运动}的\textbf{金牌大户}。
    \end{itemize}
}

\section{组织机构}

\state{国际游泳联合会}{
    \begin{itemize}
        \item 游泳的\textbf{最高组织机构}是\textbf{国际游泳联合会}（FINA）；
        \item 总部设在\textbf{瑞士洛桑}。
    \end{itemize}
}
