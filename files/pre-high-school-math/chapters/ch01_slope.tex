\chapter{斜率 $K$}

平面直角坐标系中，若一条非竖直直线与 $x$ 轴正方向的夹角为 $\alpha$（$0 \le \alpha < \pi$），则定义该直线的\textbf{斜率}为
\[
k = \tan\alpha.
\]
几何上，$k$ 衡量"每向右走 $1$ 单位，$y$ 上升多少"；代数上，对直线上任意两点 $P_1(x_1,y_1)$ 与 $P_2(x_2,y_2)$（$x_1 \ne x_2$），有
\[
k = \frac{y_2 - y_1}{x_2 - x_1}.
\]

\defn{斜率}{
    一条非竖直直线 $\ell$ 的\textbf{斜率}为其倾角 $\alpha$ 的正切值 $k = \tan\alpha$。竖直直线的斜率不存在（或约定为 $\infty$）。
}

\prop{平行与垂直判据}{
    设两条直线 $\ell_1, \ell_2$ 的斜率分别为 $k_1, k_2$（均存在），则
    \[
    \ell_1 \parallel \ell_2 \iff k_1 = k_2,
    \qquad
    \ell_1 \perp \ell_2 \iff k_1 k_2 = -1.
    \]
}

\ex[两点求斜率]{
    求经过 $A(-1, 2)$ 与 $B(3, -4)$ 的直线斜率。
}

\sol{
    由两点式，
    \[
    k = \frac{-4 - 2}{3 - (-1)} = \frac{-6}{4} = -\frac{3}{2}.
    \]
}

\ex[垂直判定]{
    直线 $\ell_1: y = 2x + 1$ 与 $\ell_2: y = -\tfrac{1}{2}x + 5$ 是否垂直？
}

\sol{
    $k_1 = 2,\ k_2 = -\tfrac{1}{2}$，故 $k_1 k_2 = -1$，二者垂直。
}
